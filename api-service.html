Quick Start Example

Get started in under 5 minutes with our simple API 

// Get video information
fetch('https://api.ytd.com/api/video/info', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-API-Key': 'your_api_key_here'
  },
  body: JSON.stringify({
    url: 'https://youtube.com/watch?v=...'
  })
})
.then(res => res.json())
.then(data => {
  console.log('Title:', data.data.title);
  console.log('Formats:', data.data.formats);
  console.log('Remaining Requests:', data.remaining_requests);
});
